83class MovementSystem
84{
85 public:
86 // Intègre les vitesses en suivant le pas de tick serveur (32 ms) pour éviter les accélérations/ralentissements
87 void update(ECS &ecs, Entity e, float dt)
88 {
89 auto &pos = ecs.getComponent<Position>(e);
90 auto &vel = ecs.getComponent<Velocity>(e);
91
92 float &acc = _accumulator[e];
93 acc += dt;
94 constexpr float tickDuration = 0.032f; // 32 ms
95 while (acc >= tickDuration) {
96 pos.x += vel.vx;
97 pos.y += vel.vy;
98 acc -= tickDuration;
99 }
100 }
101
102 private:
103 std::unordered_map<Entity, float> _accumulator;
104 };
