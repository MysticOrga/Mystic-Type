15/**
16 * @brief Supported packet types exchanged between server and clients.
17 */
18enum class PacketType : uint8_t {
19 SERVER_HELLO = 1, ///< Server greeting with assigned id.
20 CLIENT_HELLO = 2, ///< Client greeting when connecting.
21 OK = 3, ///< Generic acknowledgement.
22 REFUSED = 4, ///< Connection refused or invalid request.
23 PING = 5, ///< Heartbeat ping.
24 PONG = 6, ///< Heartbeat response.
25 MESSAGE = 7, ///< Text or generic payload message.
26 PLAYER_LIST = 8, ///< Broadcast of currently connected players.
27 NEW_PLAYER = 9, ///< Notification of a newly connected player.
28 HELLO_UDP = 10, ///< UDP hello used for NAT hole punching.
29 INPUT = 11, ///< Player input payload.
30 SNAPSHOT = 12, ///< World state snapshot.
31 SHOOT = 13 ///< Player shoot command.
32};
