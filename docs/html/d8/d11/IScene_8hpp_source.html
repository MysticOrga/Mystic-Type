14/**
15 * @class IScene
16 * @brief Abstract base interface for all scene implementations
17 *
18 * Defines the interface that all scenes must implement to work with the
19 * SceneManager. A scene represents a distinct state or screen in the game,
20 * such as menu, lobby, gameplay, or end screen.
21 *
22 * @note All derived classes must implement the pure virtual methods.
23 */
24class IScene {
25public:
26 /**
27 * @brief Virtual destructor
28 */
29 virtual ~IScene() = default;
30
31 /**
32 * @brief Called when the scene becomes the active scene
33 *
34 * Use this method to initialize scene-specific resources,
35 * such as loading assets, setting up entities, or initializing state.
36 */
37 virtual void onEnter() = 0;
38
39 /**
40 * @brief Called when the scene is being deactivated
41 *
42 * Use this method to clean up resources and save any necessary state
43 * before transitioning to another scene.
44 */
45 virtual void onExit() = 0;
46
47 /**
48 * @brief Updates the scene logic
49 *
50 * Called every frame to update game logic, handle input, process
51 * network events, and update entity states.
52 *
53 * @param dt Delta time since the last frame (in seconds)
54 */
55 virtual void update(float dt) = 0;
56
57 /**
58 * @brief Renders the scene
59 *
60 * Called every frame to render all visual elements of the scene.
61 * This should include drawing backgrounds, entities, UI elements, etc.
62 *
63 * @param dt Delta time since the last frame (in seconds)
64 */
65 virtual void render(float dt) = 0;
66
67 /**
68 * @brief Sets the scene manager reference (optional)
69 *
70 * Called by SceneManager to provide a reference to itself.
71 * This allows scenes to transition to other scenes.
72 *
73 * @param sceneManager Pointer to the SceneManager instance
74 *
75 * @note Only override this if your scene needs to change scenes.
76 * The default implementation does nothing.
77 */
78 virtual void setSceneManager(SceneManager* sceneManager) {}
79};
