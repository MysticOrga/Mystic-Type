61void GameScene::syncEntities(const std::vector<PlayerState>& players)
62{
63 int myId = _net.getPlayerId();
64
65 for (const auto& p : players) {
66 if (_entities.size() >= 4 && _entities.find(p.id) == _entities.end())
67 continue;
68
69 float clientX = static_cast<float>(p.x);
70 float clientY = static_cast<float>(p.y);
71
72 auto it = _entities.find(p.id);
73 if (it == _entities.end()) {
74 Entity ent = createPlayerEntity(clientX, clientY);
75 _entities[p.id] = ent;
76 } else {
77 if (p.id == myId) {
78 continue;
79 }
80
81 auto& pos = _ecs.getComponent<Position>(it->second);
82 auto& vel = _ecs.getComponent<Velocity>(it->second);
83
84 float smoothingFactor = 0.1f;
85 vel.vx = (clientX - pos.x) * smoothingFactor;
86 vel.vy = (clientY - pos.y) * smoothingFactor;
87 }
88 }
89}

91void GameScene::syncBullets(const std::vector<BulletState>& bullets)
92{
93 std::unordered_set<int> liveIds;
94 for (const auto& b : bullets) {
95 liveIds.insert(b.id);
96 float clientX = static_cast<float>(b.x);
97 float clientY = static_cast<float>(b.y);
98 float serverVx = static_cast<float>(b.vx);
99 float serverVy = static_cast<float>(b.vy);
100 auto it = _bulletEntities.find(b.id);
101 if (it == _bulletEntities.end()) {
102 Entity ent = createBulletEntity(clientX, clientY, serverVx, serverVy);
103 _bulletEntities[b.id] = ent;
104 } else {
105 auto& pos = _ecs.getComponent<Position>(it->second);
106 auto& vel = _ecs.getComponent<Velocity>(it->second);
107
108 vel.vx = serverVx;
109 vel.vy = serverVy;
110 }
111 }
112 for (auto& kv : _bulletEntities) {
113 if (liveIds.find(kv.first) == liveIds.end()) {
114 auto& pos = _ecs.getComponent<Position>(kv.second);
115 auto& vel = _ecs.getComponent<Velocity>(kv.second);
116 pos.x = -1000.0f;
117 pos.y = -1000.0f;
118 vel.vx = 0.0f;
119 vel.vy = 0.0f;
120 }
121 }
122}

124void GameScene::syncMonsters(const std::vector<MonsterState>& monsters)
125{
126 std::unordered_set<int> liveIds;
127 for (const auto& m : monsters) {
128 liveIds.insert(m.id);
129 float clientX = static_cast<float>(m.x);
130 float clientY = static_cast<float>(m.y);
131 auto it = _monsterEntities.find(m.id);
132 if (it == _monsterEntities.end()) {
133 Entity ent = createMonsterEntity(clientX, clientY, m.type);
134 _monsterEntities[m.id] = ent;
135 } else {
136 auto& pos = _ecs.getComponent<Position>(it->second);
137 auto& rect = _ecs.getComponent<RectangleComponent>(it->second);
138 const float smoothing = 0.25f;
139 pos.x += (clientX - pos.x) * smoothing;
140 pos.y += (clientY - pos.y) * smoothing;
141 rect.color = (m.type == 1) ? BLUE : RED;
142 }
143 }
144 for (auto& kv : _monsterEntities) {
145 if (liveIds.find(kv.first) == liveIds.end()) {
146 auto& pos = _ecs.getComponent<Position>(kv.second);
147 pos.x = -1000.0f;
148 pos.y = -1000.0f;
149 }
150 }
151}

153void GameScene::processNetworkEvents()
154{
155 _net.pollPackets();
156 for (const auto& ev : _net.getEvents()) {
157 if (ev == "PLAYER_LIST" || ev == "NEW_PLAYER") {
158 for (const auto& p : _net.getLastPlayerList())
159 _state.upsertPlayer(p.id, p.x, p.y);
160 } else if (ev == "SNAPSHOT") {
161 for (const auto& p : _net.getLastSnapshot())
162 _state.upsertPlayer(p.id, p.x, p.y);
163 for (const auto& m : _net.getLastSnapshotMonsters())
164 _state.upsertMonster(m.id, m.x, m.y, m.hp, m.type);
165 }
166 }
167 _net.clearEvents();
168}

170void GameScene::updateEntities(float dt)
171{
172 syncEntities(_state.listPlayers());
173 syncBullets(_net.getLastSnapshotBullets());
174 syncMonsters(_net.getLastSnapshotMonsters());
175
176 int myId = _net.getPlayerId();
177
178 if (_entities.find(myId) != _entities.end()) {
179 Entity myEntity = _entities[myId];
180 const auto& myPos = _ecs.getComponent<Position>(myEntity);
181 auto& myVel = _ecs.getComponent<Velocity>(myEntity);
182
183 _inputSystem.update(_net, myPos, myVel);
184 }
185
186 for (const auto& kv : _entities) {
187 _movementSystem.update(_ecs, kv.second, dt);
188 }
189 for (const auto& kv : _bulletEntities) {
190 _movementSystem.update(_ecs, kv.second, dt);
191 }
192
193 if (_entities.find(myId) != _entities.end()) {
194 Entity myEntity = _entities[myId];
195 auto& pos = _ecs.getComponent<Position>(myEntity);
196
197 if (pos.x < 0) pos.x = 0;
198 if (pos.y < 0) pos.y = 0;
199 if (pos.x > 255) pos.x = 255;
200 if (pos.y > 255) pos.y = 255;
201 }
202}
