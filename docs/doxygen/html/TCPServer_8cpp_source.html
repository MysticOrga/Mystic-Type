23 std::string sanitizePseudo(const std::string &raw)
24 {
25 std::string out;
26 out.reserve(raw.size());
27 for (char c : raw) {
28 if ((c >= 'a' && c <= 'z') || (c >= 'A' && c <= 'Z')
29 || (c >= '0' && c <= '9') || c == '_' || c == '-') {
30 out.push_back(c);
31 }
32 if (out.size() >= 12)
33 break;
34 }
35 return out;
36 }

102TCPServer::RecvResult TCPServer::receivePacket(socket_t fd, Packet &packet, std::vector<uint8_t> &recvBuffer)
103{
104 if (fd == INVALID_SOCKET_FD) {
105 return RecvResult::Disconnected;
106 }
107
108 uint8_t tmp[BUFFER_SIZE]{};
109 ssize_t n = readFd(fd, tmp, sizeof(tmp));
110 if (n <= 0) {
111 return RecvResult::Disconnected;
112 }
113
114 auto status = Protocol::consumeChunk(tmp, static_cast<std::size_t>(n), recvBuffer, packet);
115 if (status == Protocol::StreamStatus::Ok)
116 return RecvResult::Ok;
117 if (status == Protocol::StreamStatus::Incomplete)
118 return RecvResult::Incomplete;
119 return RecvResult::Disconnected;
120}

165void TCPServer::acceptNewClient()
166{
167 sockaddr_in addr{};
168 socket_t clientFd = _serverSocket.acceptClient(addr);
169
170 if (clientFd == INVALID_SOCKET_FD)
171 return;
172 char flag = 1;
173 setsockopt(clientFd, IPPROTO_TCP, TCP_NODELAY, &flag, sizeof(flag));
174
175 int slot = INVALID_SOCKET_FD;
176 for (size_t i = 0; i < _clients.size(); i++) {
177 if (_clients[i].fd == INVALID_SOCKET_FD) {
178 slot = i;
179 break;
180 }
181 }
182
183 if (slot == INVALID_SOCKET_FD) {
184 sendPacket(clientFd, makeStringPacket(PacketType::REFUSED, "FULL"));
185 closeFd(clientFd);
186 std::cout << "[SERVER] refused new client (FULL)\n";
187 return;
188 }
189
190 _clients[slot].fd = clientFd;
191 _clients[slot].addr = addr;
192 _clients[slot].id = _nextId++;
193 _clients[slot].posX = static_cast<uint8_t>(slot);
194 _clients[slot].posY = 0;
195 _clients[slot].hp = kDefaultPlayerHp;
196 _sessions.addSession(_clients[slot].id, clientFd, addr, getCurrentTime());
197 _clients[slot].handshakeStart = getCurrentTime();
198 sendPacket(_clients[slot].fd, makeStringPacket(PacketType::SERVER_HELLO, "R-Type Server"));
199
200 std::cout << "[SERVER] client " << _clients[slot].id << " connected (awaiting CLIENT_HELLO)\n";
201}

203void TCPServer::processClientData(Client &client)
204{
205 Packet packet;
206 auto res = receivePacket(client.fd, packet, client.recvBuffer);
207 if (res == RecvResult::Disconnected) {
208 std::cout << "[SERVER] client " << client.id << " disconnected\n";
209 if (client.handshakeDone && !client.lobbyCode.empty()) {
210 std::string sys = "SYS:" + client.pseudo + " disconnected";
211 broadcastToLobby(client.lobbyCode, makeStringPacket(PacketType::MESSAGE, sys));
212 }
213 resetClient(client);
214 return;
215 }
216
217 if (res == RecvResult::Incomplete) {
218 return;
219 }
220
221 if (!client.handshakeDone) {
222 std::string payloadStr(packet.payload.begin(), packet.payload.end());
223 if (packet.type != PacketType::CLIENT_HELLO || payloadStr.rfind("toto", 0) != 0) {
224 sendPacket(client.fd, makeStringPacket(PacketType::REFUSED, "BAD_HANDSHAKE"));
225 resetClient(client);
226 return;
227 }
228 std::string pseudo;
229 auto sep = payloadStr.find('|');
230 if (sep != std::string::npos && sep + 1 < payloadStr.size()) {
231 pseudo = payloadStr.substr(sep + 1);
232 }
233 pseudo = sanitizePseudo(pseudo);
234 if (pseudo.empty()) {
235 pseudo = "Player" + std::to_string(client.id);
236 }
237 sendPacket(client.fd, makeIdPacket(PacketType::OK, client.id));
238 client.handshakeDone = true;
239 client.pseudo = pseudo;
240 client.lastPongTime = getCurrentTime();
241 _sessions.setPseudo(client.id, pseudo);
242 std::cout << "[SERVER] client " << client.id << " handshake done (awaiting lobby selection)\n";
243 return;
244 }
245
246 if (packet.type == PacketType::PONG) {
247 client.lastPongTime = getCurrentTime();
248 _sessions.updatePong(client.id, client.lastPongTime);
249 std::cout << "[SERVER] Received PONG from client " << client.id << std::endl;
250 return;
251 }
252
253 if (packet.type == PacketType::CREATE_LOBBY || packet.type == PacketType::JOIN_LOBBY) {
254 handleLobbyPacket(client, packet);
255 return;
256 }
257
258 if (packet.type == PacketType::MESSAGE) {
259 std::cerr << "message\n";
260 std::string lobbyCode = client.lobbyCode;
261 if (lobbyCode.empty()) {
262 auto lobbyOpt = _sessions.getLobbyCode(client.id);
263 if (lobbyOpt.has_value())
264 lobbyCode = *lobbyOpt;
265 }
266 std::cout << "[SERVER] CHAT from id=" << client.id
267 << " lobby=" << (lobbyCode.empty() ? "?" : lobbyCode) << "\n";
268 if (!lobbyCode.empty()) {
269 std::string text(packet.payload.begin(), packet.payload.end());
270 std::string clean;
271 clean.reserve(text.size());
272 for (char c : text) {
273 if (c >= 32 && c <= 126) {
274 clean.push_back(c);
275 }
276 if (clean.size() >= 120)
277 break;
278 }
279 if (!clean.empty()) {
280 std::string name = client.pseudo;
281 if (name.empty()) {
282 auto pseudoOpt = _sessions.getPseudo(client.id);
283 if (pseudoOpt.has_value())
284 name = *pseudoOpt;
285 }
286 if (name.empty()) {
287 name = "Player" + std::to_string(client.id);
288 }
289 std::string msg = "CHAT:" + name + ": " + clean;
290 int recipients = 0;
291 for (auto &c : _clients) {
292 if (c.fd != -1 && c.handshakeDone && c.lobbyCode == lobbyCode)
293 recipients++;
294 }
295 std::cout << "[SERVER] Broadcast lobby=" << lobbyCode << " recipients=" << recipients
296 << " msg=\"" << msg << "\"\n";
297 broadcastToLobby(lobbyCode, makeStringPacket(PacketType::MESSAGE, msg));
298 }
299 } else {
300 std::string text(packet.payload.begin(), packet.payload.end());
301 std::string clean;
302 clean.reserve(text.size());
303 for (char c : text) {
304 if (c >= 32 && c <= 126) {
305 clean.push_back(c);
306 }
307 if (clean.size() >= 120)
308 break;
309 }
310 if (!clean.empty()) {
311 std::string name = client.pseudo.empty() ? ("Player" + std::to_string(client.id)) : client.pseudo;
312 std::string msg = "CHAT:" + name + ": " + clean;
313 int recipients = 0;
314 for (auto &c : _clients) {
315 if (c.fd != -1 && c.handshakeDone)
316 recipients++;
317 }
318 std::cout << "[SERVER] Broadcast ALL recipients=" << recipients << " msg=\"" << msg << "\"\n";
319 for (auto &c : _clients) {
320 if (c.fd != -1 && c.handshakeDone)
321 sendPacket(c.fd, makeStringPacket(PacketType::MESSAGE, msg));
322 }
323 }
324 }
325 return;
326 }
327
328 std::string payloadStr(packet.payload.begin(), packet.payload.end());
329 std::cout << "[SERVER] (" << client.id << ") packet type " << static_cast<int>(packet.type)
330 << " payload: " << payloadStr << std::endl;
331}

351void TCPServer::checkHeartbeat()
352{
353 long now = getCurrentTime();
354
355 for (auto &c : _clients) {
356 if (c.fd == -1)
357 continue;
358
359 if (!c.handshakeDone) {
360 if (now - c.handshakeStart > 3) {
361 sendPacket(c.fd, makeStringPacket(PacketType::REFUSED, "TIMEOUT"));
362 resetClient(c);
363 }
364 continue;
365 }
366 if (now - c.lastPongTime > 10) {
367 std::cout << "[SERVER] Client " << c.id << " timed out (no PONG for " << (now - c.lastPongTime) << "s)" << std::endl;
368 if (c.handshakeDone && !c.lobbyCode.empty()) {
369 std::string sys = "SYS:" + c.pseudo + " disconnected";
370 broadcastToLobby(c.lobbyCode, makeStringPacket(PacketType::MESSAGE, sys));
371 }
372 resetClient(c);
373 }

377void TCPServer::processIpcMessages()
378{
379 for (auto &kv : _lobbies) {
380 auto &ipc = kv.second.ipc;
381 if (!ipc)
382 continue;
383 while (ipc)
384 {
385 auto msgOpt = ipc->recv(0);
386 if (!msgOpt.has_value())
387 break;
388 const std::string &msg = *msgOpt;
389 if (msg.rfind("BOSS_DEAD:", 0) == 0) {
390 std::string lobbyCode = msg.substr(10);
391 if (!lobbyCode.empty()) {
392 broadcastToLobby(lobbyCode, makeStringPacket(PacketType::MESSAGE, "SYS:Boss defeated - win"));
393 }
394 continue;
395 }
396 if (msg.rfind("NO_PLAYERS:", 0) == 0) {
397 std::string lobbyCode = msg.substr(11);
398 if (!lobbyCode.empty()) {
399 broadcastToLobby(lobbyCode, makeStringPacket(PacketType::MESSAGE, "SYS:No players left - game over"));
400 auto lobbyIt = _lobbies.find(lobbyCode);
401 if (lobbyIt != _lobbies.end()) {
402 lobbyIt->second.udpPort = 0;
403 if (lobbyIt->second.ipc) {
404 lobbyIt->second.ipc->close();
405 lobbyIt->second.ipc.reset();
406 }
407 }
408 if (_childMgr) {
409 _childMgr->forget(lobbyCode);
410 }
411 }
412 break;
413 }
414 if (msg.rfind("BOSS:", 0) == 0) {
415 std::string lobbyCode = msg.substr(5);
416 if (!lobbyCode.empty()) {
417 broadcastToLobby(lobbyCode, makeStringPacket(PacketType::MESSAGE, "SYS:Boss spawned"));
418 }
419 continue;
420 }
421 if (msg.rfind("DEAD:", 0) == 0) {
422 int id = 0;
423 try {
424 id = std::stoi(msg.substr(5));
425 } catch (const std::exception &) {
426 continue;
427 }
428 if (id <= 0)
429 continue;
430 for (auto &c : _clients) {
431 if (c.fd != -1 && c.id == id) {
432 if (!c.lobbyCode.empty()) {
433 std::string sys = "SYS:" + c.pseudo + " died";
434 broadcastToLobby(c.lobbyCode, makeStringPacket(PacketType::MESSAGE, sys));
435 }
436 sendPacket(c.fd, makeStringPacket(PacketType::MESSAGE, "DEAD"));
437 removeFromLobby(c);
438 break;
439 }
440 }
441 }
442 }
443 }

446void TCPServer::run()
447{
448 long lastPing = getCurrentTime();
449
450 while (true) {
451 processIpcMessages();
452 long now = getCurrentTime();
453 if (now - lastPing >= 5) {
454 sendPingToAll();
455 checkHeartbeat();
456 lastPing = now;
457 }
458
459 fd_set readfds;
460 FD_ZERO(&readfds);
461
462 int maxFd = _serverSocket.getSocketFd();
463 FD_SET(_serverSocket.getSocketFd(), &readfds);
464
465 for (auto &c : _clients) {
466 if (c.fd != -1) {
467 FD_SET(c.fd, &readfds);
468 maxFd = maxFd > c.fd ? maxFd : c.fd;
469 }
470 }
471 for (const auto &kv : _lobbies) {
472 if (!kv.second.ipc)
473 continue;
474 int ipcFd = kv.second.ipc->fd();
475 if (ipcFd != -1) {
476 FD_SET(ipcFd, &readfds);
477 maxFd = std::max(maxFd, ipcFd);
478 }
479 }
480
481 struct timeval tv;
482 tv.tv_sec = 0;
483 tv.tv_usec = 1000;
484
485 int activity = pollSockets(readfds, maxFd, tv);
486 if (activity < 0) {
487 continue;
488 }
489
490 if (activity == 0) {
491 std::this_thread::sleep_for(std::chrono::milliseconds(1));
492 continue;
493 }
494
495 for (const auto &kv : _lobbies) {
496 if (!kv.second.ipc)
497 continue;
498 int ipcFd = kv.second.ipc->fd();
499 if (ipcFd != -1 && FD_ISSET(ipcFd, &readfds)) {
500 processIpcMessages();
501 break;
502 }
503 }
504
505 if (FD_ISSET(_serverSocket.getSocketFd(), &readfds)) {
506 acceptNewClient();
507 }
508
509 for (auto &c : _clients) {
510 if (c.fd != -INVALID_SOCKET_FD && FD_ISSET(c.fd, &readfds)) {
511 processClientData(c);
512 }
513 }
514 }
515}

549Packet TCPServer::buildPlayerListPacket(const std::string &lobbyCode) const
550{
551 std::vector<uint8_t> payload;
552 payload.reserve(1 + _clients.size() * 5);
553
554 payload.push_back(0); // placeholder for count
555 uint8_t count = 0;
556
557 for (const auto &c : _clients) {
558 if (c.fd == -1 || !c.handshakeDone || c.lobbyCode != lobbyCode)
559 continue;
560 ++count;
561 payload.push_back(static_cast<uint8_t>((c.id >> 8) & 0xFF));
562 payload.push_back(static_cast<uint8_t>(c.id & 0xFF));
563 payload.push_back(c.posX);
564 payload.push_back(c.posY);
565 payload.push_back(c.hp);
566 }
567
568 payload[0] = count;
569 return Packet(PacketType::PLAYER_LIST, payload);
570}

605std::string TCPServer::generateLobbyCode()
606{
607 static const char chars[] = "ABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789";
608 static std::mt19937 rng(static_cast<unsigned long>(std::chrono::steady_clock::now().time_since_epoch().count()));
609 std::uniform_int_distribution<int> dist(0, static_cast<int>(sizeof(chars) - 2));
610
611 std::string code(6, 'A');
612 do {
613 for (auto &ch : code) {
614 ch = chars[dist(rng)];
615 }
616 } while (_lobbies.find(code) != _lobbies.end());
617 return code;
618}

633 }
634 _clientLobby.erase(client.id);
635 _sessions.setLobbyCode(client.id, "");
636 refreshLobby(code);

669bool TCPServer::assignLobby(Client &client, const std::string &code, bool createIfMissing, bool isPublic, bool allowFull)
670{
671 removeFromLobby(client);
672
673 auto it = _lobbies.find(code);
674 if (it == _lobbies.end()) {
675 if (!createIfMissing)
676 return false;
677 _lobbies[code] = LobbyInfo{isPublic, {}, 0};
678 it = _lobbies.find(code);
679 ensureLobbyProcess(code, isPublic);

683 return false;

685 it->second.players.push_back(client.id);
686 client.lobbyCode = code;
687 _clientLobby[client.id] = code;
688 _sessions.setLobbyCode(client.id, code);
689 if (it->second.udpPort == 0) {
690 ensureLobbyProcess(code, isPublic);
691 }
692 return true;
693}

702void TCPServer::handleLobbyPacket(Client &client, const Packet &packet)
703{
704 if (!client.handshakeDone)
705 return;
706
707 if (packet.type == PacketType::CREATE_LOBBY) {
708 std::cout << "[SERVER] client " << client.id << " requested CREATE_LOBBY\n";
709 std::string code = generateLobbyCode();
710 if (!assignLobby(client, code, true, false, true)) {
711 sendPacket(client.fd, makeLobbyPacket(PacketType::LOBBY_ERROR, "INVALID_STATE"));
712 return;
713 }
