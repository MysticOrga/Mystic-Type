18UDPGameServer::UDPGameServer(uint16_t port, SessionManager &sessions, long long snapshotIntervalMs, std::string lobbyCode)
19 : _port(port), _snapshotIntervalMs(snapshotIntervalMs), _sessions(sessions), _expectedLobby(std::move(lobbyCode))
20{
21 if (!_socket.bindTo(port))
22 {
23 throw std::runtime_error("Failed to bind UDP socket");
24 }
25 std::cout << logPrefix() << "Listening on port " << port << std::endl;
26 if (!_expectedLobby.empty()) {
27 _worlds[_expectedLobby] = GameWorld{};
28 _worlds[_expectedLobby].setLogPrefix(logPrefix());
29 }
30 _sessions.setOnRemove([this](int id) {
31 auto itLobby = _playerLobby.find(id);
32 if (itLobby != _playerLobby.end()) {
33 auto worldIt = _worlds.find(itLobby->second);
34 if (worldIt != _worlds.end()) {
35 worldIt->second.removePlayer(id);
36 if (worldIt->second.players().empty()) {
37 _worlds.erase(worldIt);
38 }
39 }
40 _playerLobby.erase(itLobby);
41 }
42 });
43}

75void UDPGameServer::handleHello(const Packet &packet, const sockaddr_in &from)
76{
77 if (packet.payload.size() < 2) {
78 std::cerr << logPrefix() << "HELLO_UDP payload too small\n";
79 return;
80 }
81 int id = (packet.payload[0] << 8) | packet.payload[1];
82 uint8_t x = packet.payload.size() >= 3 ? packet.payload[2] : 0;
83 uint8_t y = packet.payload.size() >= 4 ? packet.payload[3] : 0;
84
85 // Use expected lobby when provided; fallback to session info or PUBLIC.
86 std::string lobbyCode = !_expectedLobby.empty() ? _expectedLobby : "PUBLIC";
87 auto sessionOpt = _sessions.getSession(id);
88 if (sessionOpt.has_value() && !sessionOpt->lobbyCode.empty()) {
89 lobbyCode = sessionOpt->lobbyCode;
90 _sessions.setUdpAddr(id, from);
91 }
92
93 GameWorld &world = _worlds[lobbyCode];
94 world.setLogPrefix(logPrefix());
95 world.registerPlayer(id, x, y, from);
96 _playerLobby[id] = lobbyCode;
97 // Send a fresh snapshot immediately so the client sees the lobby state without waiting the next tick.
98 Packet snap = world.buildSnapshotPacket();
99 sendPacketTo(snap, from);
100 std::cout << logPrefix() << "Registered client id=" << id << " at " << static_cast<int>(x) << "," << static_cast<int>(y) << "\n";
101}

103void UDPGameServer::handleInput(const Packet &packet, const sockaddr_in &from)
104{
105 if (packet.payload.size() < 7) {
106 std::cerr << logPrefix() << "INPUT payload too small\n";
107 return;
108 }
109 int id = (packet.payload[0] << 8) | packet.payload[1];
110 // ignore client-provided position to keep authority
111 int8_t velX = static_cast<int8_t>(packet.payload[4]);
112 int8_t velY = static_cast<int8_t>(packet.payload[5]);
113 uint8_t dir = packet.payload[6];
114
115 // Rate limiting disabled here because SessionManager is not shared across processes.
116 auto lobbyIt = _playerLobby.find(id);
117 if (lobbyIt == _playerLobby.end()) {
118 if (!_expectedLobby.empty()) {
119 _playerLobby[id] = _expectedLobby;
120 } else {
121 auto sessionOpt = _sessions.getSession(id);
122 if (sessionOpt.has_value() && !sessionOpt->lobbyCode.empty()) {
123 _playerLobby[id] = sessionOpt->lobbyCode;
124 } else {
125 _playerLobby[id] = "PUBLIC";
126 }
127 }
128 lobbyIt = _playerLobby.find(id);
129 }
130
131 auto worldIt = _worlds.find(lobbyIt->second);
132 if (worldIt == _worlds.end()) {
133 worldIt = _worlds.emplace(lobbyIt->second, GameWorld{}).first;
134 }
135 worldIt->second.setLogPrefix(logPrefix());
136
137 worldIt->second.updateInput(id, velX, velY, dir, from);
138}

174void UDPGameServer::handlePacket(const Packet &packet, const sockaddr_in &from)
175{
176 switch (packet.type)
177 {
178 case PacketType::HELLO_UDP:
179 handleHello(packet, from);
180 break;
181 case PacketType::INPUT:
182 handleInput(packet, from);
183 break;
184 case PacketType::SHOOT:
185 handleShoot(packet);
186 break;
187 case PacketType::PING_UDP:
188 sendPacketTo(Packet(PacketType::PONG_UDP, packet.payload), from);
189 break;
190 default:
191 break;
192 }
193}

195void UDPGameServer::run()
196{
197 _running = true;
198 _networkThread = std::thread(&UDPGameServer::networkLoop, this);
199
200 _lastSnapshotMs = nowMs();
201 _lastTickMs = _lastSnapshotMs;
202
203 while (_running) {
204 {
205 std::lock_guard<std::mutex> lock(_queueMutex);
206 while (!_incoming.empty()) {
207 auto item = _incoming.front();
208 _incoming.pop();
209 handlePacket(item.pkt, item.from);
210 }
211 }
212
213 long long now = nowMs();
214 if (now - _lastTickMs >= _tickIntervalMs)
215 {
216 updateSimulation(now, now - _lastTickMs);
217 _lastTickMs = now;
218 }
219
220 if (now - _lastSnapshotMs >= _snapshotIntervalMs)
221 {
222 broadcastSnapshot();
223 _lastSnapshotMs = now;
224 }
225 _ipc->send("RUNNING");
226 std::this_thread::sleep_for(std::chrono::milliseconds(1));
227 }
228
229 if (_networkThread.joinable()) {
230 _networkThread.join();
231 }
232}

234void UDPGameServer::updateSimulation(long long nowMs, long long deltaMs)
235{
236 std::vector<std::string> emptyWorlds;
237 for (auto &kv : _worlds) {
238 kv.second.tick(nowMs, deltaMs);
239 if (_ipc && kv.second.takeBossSpawned()) {
240 _ipc->send("BOSS:" + kv.first);
241 }
242 if (_ipc && kv.second.takeBossDefeated()) {
243 _ipc->send("BOSS_DEAD:" + kv.first);
244 _running = false;
245 }
246 std::vector<int> toRemove;
247 for (const auto &p : kv.second.players()) {
248 if (p.second.hp == 0) {
249 toRemove.push_back(p.first);
250 }
251 }
252 for (int id : toRemove) {
253 if (_ipc) {
254 _ipc->send("DEAD:" + std::to_string(id));
255 }
256 kv.second.removePlayer(id);
257 _playerLobby.erase(id);
258 }
259 if (kv.second.takeNoPlayers()) {
260 emptyWorlds.push_back(kv.first);
261 }
262 }
263 for (const auto &code : emptyWorlds) {
264 _worlds.erase(code);
265 if (_ipc) {
266 _ipc->send("NO_PLAYERS:" + code);
267 }
268 if (!_expectedLobby.empty() && _expectedLobby == code) {
269 _running = false;
270 }
271 }
272 if (_expectedLobby.empty() && _worlds.empty()) {
273 _running = false;
274 }
275}

277void UDPGameServer::networkLoop()
278{
279 uint8_t buffer[UDP_BUFFER_SIZE]{};
280 while (_running) {
281 ssize_t n = _socket.readByte(reinterpret_cast<char *>(buffer), sizeof(buffer));
282 if (n > 0) {
283 try {
284 Packet packet = Packet::deserialize(buffer, static_cast<size_t>(n));
285 Incoming inc;
286 inc.pkt = std::move(packet);
287 inc.from = _socket.getSenderAddr();
288 std::lock_guard<std::mutex> lock(_queueMutex);
289 _incoming.push(std::move(inc));
290 } catch (const std::exception &e) {
291 std::cerr << logPrefix() << "Failed to parse packet: " << e.what() << "\n";
292 }
293 } else {
294 std::this_thread::sleep_for(std::chrono::milliseconds(1));
295 }
296 }
297}
