124

130 }

165 {
166 auto &pos = ecs.getComponent<Position>(e);
167 auto &vel = ecs.getComponent<Velocity>(e);
168
169 float &acc = _accumulator[e];
170 acc += dt;
171 constexpr float tickDuration = 0.032f;
172 while (acc >= tickDuration)
173 {
174 pos.x += vel.vx;
175 pos.y += vel.vy;
176 acc -= tickDuration;
177 }
178 }
