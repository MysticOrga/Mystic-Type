11std::vector<uint8_t> Protocol::frameTcp(const Packet &packet)
12{
13 std::vector<uint8_t> payload = packet.serialize();
14 if (payload.size() > UINT16_MAX) {
15 throw std::runtime_error("Payload too large for framing");
16 }
17 uint16_t len = static_cast<uint16_t>(payload.size());
18 std::vector<uint8_t> framed;
19 framed.reserve(payload.size() + 2);
20 framed.push_back(static_cast<uint8_t>(len >> 8));
21 framed.push_back(static_cast<uint8_t>(len & 0xFF));
22 framed.insert(framed.end(), payload.begin(), payload.end());
23 return framed;
24}

26Protocol::StreamStatus Protocol::extractFromBuffer(std::vector<uint8_t> &recvBuffer, Packet &out)
27{
28 while (recvBuffer.size() >= 2) {
29 uint16_t len = (static_cast<uint16_t>(recvBuffer[0]) << 8) | recvBuffer[1];
30 if (recvBuffer.size() < 2 + len)
31 return StreamStatus::Incomplete;
32
33 std::vector<uint8_t> pktData(recvBuffer.begin() + 2, recvBuffer.begin() + 2 + len);
34 recvBuffer.erase(recvBuffer.begin(), recvBuffer.begin() + 2 + len);
35 try {
36 out = Packet::deserialize(pktData.data(), pktData.size());
37 return StreamStatus::Ok;
38 } catch (const std::exception &) {
39 // malformed packet, skip and continue parsing
40 continue;
41 }
42 }
43 return StreamStatus::Incomplete;
44}
