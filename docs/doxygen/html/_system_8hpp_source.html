113class MovementSystem
114{
115 public:
116 // Intègre les vitesses en suivant le pas de tick serveur (32 ms) pour éviter les accélérations/ralentissements
117 void update(ECS &ecs, Entity e, float dt)
118 {
119 auto &pos = ecs.getComponent<Position>(e);
120 auto &vel = ecs.getComponent<Velocity>(e);
121
122 float &acc = _accumulator[e];
123 acc += dt;
124 constexpr float tickDuration = 0.032f; // 32 ms
125 while (acc >= tickDuration)
126 {
127 pos.x += vel.vx;
128 pos.y += vel.vy;
129 acc -= tickDuration;
130 }
131 }
132
133 private:
134 std::unordered_map<Entity, float> _accumulator;
135};
