130void GraphicClient::syncEntities(const std::vector<PlayerState> &players)
131{
132 int myId = _net.getPlayerId();
133
134 std::unordered_set<int> liveIds;
135 for (const auto &p : players)
136 {
137 liveIds.insert(p.id);
138 if (_entities.size() >= 4 && _entities.find(p.id) == _entities.end())
139 continue;
140
141 float clientX = static_cast<float>(p.x);
142 float clientY = static_cast<float>(p.y);
143
144 auto it = _entities.find(p.id);
145 if (it == _entities.end())
146 {
147 Entity ent = createPlayerEntity(clientX, clientY);
148 _entities[p.id] = ent;
149 }
150 else
151 {
152 auto &pos = _ecs.getComponent<Position>(it->second);
153 auto &vel = _ecs.getComponent<Velocity>(it->second);
154
155 float dx = clientX - pos.x;
156 float dy = clientY - pos.y;
157 float dist2 = dx * dx + dy * dy;
158
159 if (p.id == myId)
160 {
161 pos.x = clientX;
162 pos.y = clientY;
163 vel.vx = 0.0f;
164 vel.vy = 0.0f;
165 }
166 else
167 {
168 const float smooth = 0.25f;
169 if (dist2 > 100.0f)
170 {
171 pos.x = clientX;
172 pos.y = clientY;
173 }
174 else
175 {
176 pos.x += dx * smooth;
177 pos.y += dy * smooth;
178 }
179 vel.vx = 0.0f;
180 vel.vy = 0.0f;
181 }
182 }
183 }
184
185 // Remove entities that are no longer in the snapshot/player list
186 for (auto it = _entities.begin(); it != _entities.end();)
187 {
188 if (liveIds.find(it->first) == liveIds.end())
189 {
190 it = _entities.erase(it);
191 }
192 else
193 {
194 ++it;
195 }
196 }

859void GraphicClient::gameLoop()
860{
861 while (!_window.shouldClose())
862 {
863 float dt = _window.getFrameTime();
864 // std::cerr << "before processing network event" << i << std::endl;
865 processNetworkEvents();
866 if (_pendingReturnToLobby) {
867 if (std::chrono::steady_clock::now() >= _returnToLobbyAt) {
868 std::cerr << "[CLIENT] " << _pendingReturnReason << "\n";
869 _net.disconnectUdp();
870 _forceExit = true;
871 _restartToMenu = true;
872 _pendingReturnToLobby = false;
873 _pendingReturnReason.clear();
874 }
875 }
876 if (_forceExit)
877 return;
878 // If PING send failed, drop to avoid server timeout
879 for (const auto &ev : _net.getEvents())
880 {
881 if (ev == "PING_SEND_FAIL")
882 {
883 std::cerr << "[CLIENT] Failed to send PONG, disconnecting\n";
884 _net.disconnect();
885 return;
886 }
887 }
888 _net.clearEvents();
889
890 // Proactive keepalive to avoid timeout when idle in game
891 auto nowKeep = std::chrono::steady_clock::now();
892 if (std::chrono::duration_cast<std::chrono::seconds>(nowKeep - _lastKeepAlive).count() >= 4)
893 {
894 _net.sendPong();
895 _lastKeepAlive = nowKeep;
896 }
897
898 if (!_udpReady)
899 {
900 auto nowHello = std::chrono::steady_clock::now();
901 if (std::chrono::duration_cast<std::chrono::seconds>(nowHello - _lastHello).count() >= 1)
902 {
903 _net.sendHelloUdp(0, 0);
904 _lastHello = nowHello;
905 }
906 }
907 else
908 {
909 auto nowPing = std::chrono::steady_clock::now();
910 if (std::chrono::duration_cast<std::chrono::seconds>(nowPing - _lastUdpPing).count() >= 1)
911 {
912 _net.sendUdpPing();
913 _lastUdpPing = nowPing;
914 }
915 }
916
917 if (!_chatActive && Raylib::Input::isKeyPressed(KEY_ESCAPE))
918 {
919 _net.disconnectUdp();
920 _forceExit = true;
921 _restartToMenu = true;
922 }
923 else if (!_chatActive && Raylib::Input::isKeyPressed(KEY_ENTER))
924 {
925 _chatActive = true;
926 _chatInput.clear();
927 }
928 else if (_chatActive)
929 {
930 int key = GetCharPressed();
931 while (key > 0)
932 {
933 if (key >= 32 && key <= 126 && _chatInput.size() < 120)
934 {
935 _chatInput.push_back(static_cast<char>(key));
936 }
937 key = GetCharPressed();
938 }
939 if (Raylib::Input::isKeyPressed(KEY_BACKSPACE) && !_chatInput.empty())
940 _chatInput.pop_back();
941 if (Raylib::Input::isKeyPressed(KEY_ESCAPE))
942 {
943 _chatActive = false;
944 }
945 else if (Raylib::Input::isKeyPressed(KEY_ENTER))
946 {
947 if (!_chatInput.empty()) {
948 std::string name = _localPseudo.empty() ? "Me" : _localPseudo;
949 _chatLog.push_back(name + ": " + _chatInput);
950 if (_chatLog.size() > 8)
951 _chatLog.erase(_chatLog.begin());
952 _lastChatSent = _chatInput;
953 if (!_net.sendChat(_chatInput)) {
954 _chatLog.push_back("[SYS] sendChat failed");
955 if (_chatLog.size() > 8)
956 _chatLog.erase(_chatLog.begin());
957 }
958 }
959 _chatActive = false;
960 }
961 }
962
963 updateEntities(dt);
964 // std::cerr << "after update entities" << i << std::endl;
965 // std::cerr << "before render" << i << std::endl;
966 render(dt);
967 // std::cerr << "after render" << i << std::endl;
968 // i++;
969 }
970
971}

974{
975 while (!_window.shouldClose())
976 {
977 if (!init()) {
978 std::cerr << "[CLIENT] init failed (" << (_lastInitError.empty() ? "unknown" : _lastInitError) << "), retrying...\n";
979 std::this_thread::sleep_for(std::chrono::milliseconds(500));
980 continue;
981 }
982 gameLoop();
983 if (_restartToMenu)
984 {
985 _restartToMenu = false;
986 _forceExit = false;
987 _udpReady = false;
988 _gameAnimTimer = 0.0f;

999 _net.resetForLobby();
1000 continue;
1001 }
1002 break;
1003 }
