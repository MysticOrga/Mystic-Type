49 /**
50 * @brief Constructor for AnimatedSprite.
51 * @param path Path to sprite sheet image file.
52 * @param size Size of each frame (width, height).
53 * @param posInSheet Position of first frame in sprite sheet grid.
54 * @param maxFrames Total number of animation frames.
55 * @param frameTime Time in seconds each frame is displayed (default: 0.1).
56 * @param pos Initial screen position (default: 0, 0).
57 */
58 AnimatedSprite(const std::string &path, Vector2 size, Vector2 posInSheet, int maxFrames,
59 float frameTime = 0.1f, Vector2 pos = {0, 0})

61 _position(pos), _timer(0.0f), _currentFrame(0)

67 posInSheet.y * size.y,

73 /**

87 void setScale(float sx, float sy)

89 _scale = {sx, sy};
90 }
91
92 /**
93 * @brief Updates animation frame based on elapsed time.
94 * @param dt Delta time since last update.
95 */
96 void update(float dt)
97 {
98 _timer += dt;
99
100 if (_timer >= _frameTime)
101 {
102 _timer = 0.0f;
103 _currentFrame++;
104
105 if (_currentFrame >= _maxFrames)
106 _currentFrame = 0;
107
108 _sourceRect.x = (_positionInSpritesheet.x + _currentFrame) * _size.x;
109 }
110 }
111
112 /**
113 * @brief Renders the sprite to the screen.
114 *
115 * Draws the current animation frame centered at the sprite position.
116 */
117 void draw()
118 {
119 if (_texture) {
120 float w = _sourceRect.width * _scale.x;
121 float h = _sourceRect.height * _scale.y;
122 Vector2 centeredPos = _position;
123 centeredPos.x -= w / 2.0f;
124 centeredPos.y -= h / 2.0f;
125
126 Rectangle dest{centeredPos.x, centeredPos.y, w, h};
127 _texture->draw(_sourceRect, dest, WHITE);
128 }
129 }
130};
